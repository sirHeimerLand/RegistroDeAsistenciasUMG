CREATE TABLE Estudiantes (
    EstudianteID INT PRIMARY KEY,
    Nombre VARCHAR(100),
    Apellido VARCHAR(100),
    Email VARCHAR(100),
    FechaNacimiento DATE,
    Direccion VARCHAR(255),
    Carrera VARCHAR(100),
    CicloAcademico INT,
    Promedio FLOAT
    -- Otros campos relevantes de los estudiantes
);

-- Tabla para docentes
CREATE TABLE Docentes (
    DocenteID INT PRIMARY KEY,
    Nombre VARCHAR(100),
    Apellido VARCHAR(100),
    Email VARCHAR(100),
    Especialidad VARCHAR(100),
    Departamento VARCHAR(100)
    -- Otros campos relevantes de los docentes
);

-- Tabla para administradores
CREATE TABLE Administradores (
    AdminID INT PRIMARY KEY,
    Nombre VARCHAR(100),
    Apellido VARCHAR(100),
    Email VARCHAR(100),
    Cargo VARCHAR(100)
    -- Otros campos relevantes de los administradores
);

-- Tabla para asistencias
CREATE TABLE Asistencias (
    AsistenciaID INT PRIMARY KEY,
    EstudianteID INT,
    DocenteID INT,
    FechaHora TIMESTAMP,
    Asistio CHAR(1),
    FOREIGN KEY (EstudianteID) REFERENCES Estudiantes(EstudianteID),
    FOREIGN KEY (DocenteID) REFERENCES Docentes(DocenteID)
);

-- Tabla para permisos
CREATE TABLE Permisos (
    PermisoID INT PRIMARY KEY,
    EstudianteID INT,
    FechaInicio DATE,
    FechaFin DATE,
    Motivo VARCHAR(255),
    Estado VARCHAR(20), -- Puede ser 'Pendiente', 'Aprobado', 'Rechazado'
    AprobadoPor INT, -- Docente o administrador que aprob√≥ el permiso
    FOREIGN KEY (EstudianteID) REFERENCES Estudiantes(EstudianteID),
    FOREIGN KEY (AprobadoPor) REFERENCES Administradores(AdminID)
);


DROP TABLE Estudiantes;
DROP TABLE Docentes;
DROP TABLE Administradores;
DROP TABLE Asistencias;
DROP TABLE Permisos;
